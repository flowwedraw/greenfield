512    #[allow(clippy::clone_on_copy)]
513    #[allow(clippy::vec_init_then_push)]
514    pub fn invoke_signed_with_remaining_accounts(
515        &self,
516        signers_seeds: &[&[&[u8]]],
517        remaining_accounts: &[(
518            &'b solana_program::account_info::AccountInfo<'a>,
519            bool,
520            bool,
521        )],
522    ) -> solana_program::entrypoint::ProgramResult {
523        let mut accounts = Vec::with_capacity(14 + remaining_accounts.len());
524        if let Some(delegate_record) = self.delegate_record {
525            accounts.push(solana_program::instruction::AccountMeta::new(
526                *delegate_record.key,
527                false,
528            ));
529        } else {
530            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
531                crate::MPL_TOKEN_METADATA_ID,
532                false,
533            ));
534        }
535        accounts.push(solana_program::instruction::AccountMeta::new_readonly(
536            *self.delegate.key,
537            false,
538        ));
539        accounts.push(solana_program::instruction::AccountMeta::new(
540            *self.metadata.key,
541            false,
542        ));
543        if let Some(master_edition) = self.master_edition {
544            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
545                *master_edition.key,
546                false,
547            ));
548        } else {
549            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
550                crate::MPL_TOKEN_METADATA_ID,
551                false,
552            ));
553        }
554        if let Some(token_record) = self.token_record {
555            accounts.push(solana_program::instruction::AccountMeta::new(
556                *token_record.key,
557                false,
558            ));
559        } else {
560            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
561                crate::MPL_TOKEN_METADATA_ID,
562                false,
563            ));
564        }
565        accounts.push(solana_program::instruction::AccountMeta::new_readonly(
566            *self.mint.key,
567            false,
568        ));
569        accounts.push(solana_program::instruction::AccountMeta::new(
570            *self.token.key,
571            false,
572        ));
573        accounts.push(solana_program::instruction::AccountMeta::new_readonly(
574            *self.authority.key,
575            true,
576        ));
577        accounts.push(solana_program::instruction::AccountMeta::new(
578            *self.payer.key,
579            true,
580        ));
581        accounts.push(solana_program::instruction::AccountMeta::new_readonly(
582            *self.system_program.key,
583            false,
584        ));
585        accounts.push(solana_program::instruction::AccountMeta::new_readonly(
586            *self.sysvar_instructions.key,
587            false,
588        ));
589        if let Some(spl_token_program) = self.spl_token_program {
590            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
591                *spl_token_program.key,
592                false,
593            ));
594        } else {
595            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
596                crate::MPL_TOKEN_METADATA_ID,
597                false,
598            ));
599        }
600        if let Some(authorization_rules_program) = self.authorization_rules_program {
601            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
602                *authorization_rules_program.key,
603                false,
604            ));
605        } else {
606            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
607                crate::MPL_TOKEN_METADATA_ID,
608                false,
609            ));
610        }
611        if let Some(authorization_rules) = self.authorization_rules {
612            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
613                *authorization_rules.key,
614                false,
615            ));
616        } else {
617            accounts.push(solana_program::instruction::AccountMeta::new_readonly(
618                crate::MPL_TOKEN_METADATA_ID,
619                false,
620            ));
621        }
622        remaining_accounts.iter().for_each(|remaining_account| {
623            accounts.push(solana_program::instruction::AccountMeta {
624                pubkey: *remaining_account.0.key,
625                is_signer: remaining_account.1,
626                is_writable: remaining_account.2,
627            })
628        });
629        let mut data = DelegateStandardV1InstructionData::new()
630            .try_to_vec()
631            .unwrap();
632        let mut args = self.__args.try_to_vec().unwrap();
633        data.append(&mut args);
634
635        let instruction = solana_program::instruction::Instruction {
636            program_id: crate::MPL_TOKEN_METADATA_ID,
637            accounts,
638            data,
639        };
640        let mut account_infos = Vec::with_capacity(14 + 1 + remaining_accounts.len());
641        account_infos.push(self.__program.clone());
642        if let Some(delegate_record) = self.delegate_record {
643            account_infos.push(delegate_record.clone());
644        }
645        account_infos.push(self.delegate.clone());
646        account_infos.push(self.metadata.clone());
647        if let Some(master_edition) = self.master_edition {
648            account_infos.push(master_edition.clone());
649        }
650        if let Some(token_record) = self.token_record {
651            account_infos.push(token_record.clone());
652        }
653        account_infos.push(self.mint.clone());
654        account_infos.push(self.token.clone());
655        account_infos.push(self.authority.clone());
656        account_infos.push(self.payer.clone());
657        account_infos.push(self.system_program.clone());
658        account_infos.push(self.sysvar_instructions.clone());
659        if let Some(spl_token_program) = self.spl_token_program {
660            account_infos.push(spl_token_program.clone());
661        }
662        if let Some(authorization_rules_program) = self.authorization_rules_program {
663            account_infos.push(authorization_rules_program.clone());
664        }
665        if let Some(authorization_rules) = self.authorization_rules {
666            account_infos.push(authorization_rules.clone());
667        }
668        remaining_accounts
669            .iter()
670            .for_each(|remaining_account| account_infos.push(remaining_account.0.clone()));
671
672        if signers_seeds.is_empty() {
673            solana_program::program::invoke(&instruction, &account_infos)
674        } else {
675            solana_program::program::invoke_signed(&instruction, &account_infos, signers_seeds)
676        }
677    }

680/// Instruction builder for `DelegateStandardV1` via CPI.
681///
682/// ### Accounts:
683///
684///   0. `[writable, optional]` delegate_record
685///   1. `[]` delegate
686///   2. `[writable]` metadata
687///   3. `[optional]` master_edition
688///   4. `[writable, optional]` token_record
689///   5. `[]` mint
690///   6. `[writable]` token

692///   8. `[writable, signer]` payer
693///   9. `[]` system_program

697///   13. `[optional]` authorization_rules
